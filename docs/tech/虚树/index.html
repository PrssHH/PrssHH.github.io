P2495 [SDOI2011]消耗战

虚树是处理树上问题的一种黑科技方法，对于大量的询问，每一次都做一遍 $O(n)$ 的树上DP显然不可行，不过对于少量的询问点，发现原树只有部分信息是有用的，就是多个关键点以及他们所在的链交汇的节点，也就是各个 LCA ，所以可以在每一次询问的时候把这一些关键点抽出来，构成一个小的树，此时树上实用的信息比就变大了，时间复杂度也压下去了。

先处理处原树上的欧拉序，然后进行每一次询问的时候按照欧拉序排序，接着把还需要的点塞进来，这样就可以建成我们需要的虚树了。

利用欧拉序的入栈出站记录，模拟dfs，实现原本的树上dp的功能，得出答案。

#include <bits/stdc++.h>
using namespace std;
typedef long long lint;
inline int read() {
	int x = 0, f = 0; char c = getchar();
	for (; c < '0' || c > '9'; c = getchar()) if (c == '-') f = 1;
	for (; c >= '0' && c <= '9'; c = getchar()) x = (x << 3) + (x << 1) + (c ^ 48);
	return f ? -x : x;
}

const int maxn = 250000 + 7;
struct Edge {
	int v, w, next;
} e[maxn << 1];
int ec, head[maxn];

void add(int u, int v, int w) {
	ec++;
	e[ec].v = v;
	e[ec].w = w;
	e[ec].next = head[u];
	head[u] = ec;
}

int n, m;
int fa[maxn][20], dep[maxn], dft, dfin[maxn], dfout[maxn];
lint mi[maxn];

void dfs(int x) {
	dfin[x] = ++dft;
	for (int i = 1; i < 20; ++i) fa[x][i] = fa[fa[x][i - 1]][i - 1];
	for (int i = head[x]; i; i = e[i].next) {
		int v = e[i].v;
		if (v == fa[x][0]) continue;
		dep[v] = dep[x] + 1;
		fa[v][0] = x;
		mi[v] = min(mi[x], 1ll * e[i].w);
		dfs(v);
	}
	dfout[x] = ++dft;
}

int lca(int x, int y) {
	if (dep[x] < dep[y]) swap(x, y);
	for (int i = 0, k = dep[x] - dep[y]; k; k >>= 1, ++i) 
		if (k & 1) x = fa[x][i];
	if (x == y) return x;

	for (int i = 19; ~i; --i) 
		if (fa[x][i] != fa[y][i]) 
			x = fa[x][i], y = fa[y][i];
	return fa[x][0];
}

int tr[maxn << 2];
stack<int> s;
bool book[maxn];
lint sum[maxn];

bool cmp(int x, int y) {
	int k1 = x > 0 ? dfin[x] : dfout[-x];
	int k2 = y > 0 ? dfin[y] : dfout[-y];
	return k1 < k2;
}

int main() {
	n = read();
	for (int i = 1, u, v, w; i < n; ++i) {
		u = read(); v = read(); w = read();
		add(u, v, w); add(v, u, w);
	}
	mi[1] = 0x3f3f3f3f3f3f3f3f;
	dep[1] = 1;
	dfs(1);
	m = read();
	while (m--) {
		int k = read();
		for (int i = 1; i <= k; ++i) {
			tr[i] = read();
			book[tr[i]] = true;
			sum[tr[i]] = mi[tr[i]];
		}
		sort(tr + 1, tr + 1 + k, cmp);
		for (int i = 1; i < k; ++i) {
			int lc = lca(tr[i], tr[i + 1]);
			if (!book[lc]) tr[++k] = lc, book[lc] = true;
		}

		int nk = k;
		for (int i = 1; i <= nk; ++i) tr[++k] = -tr[i];
		if (!book[1]) { tr[++k] = 1; tr[++k] = -1; book[1] = true; }
		sort(tr + 1, tr + 1 + k, cmp);
		for (int i = 1; i <= k; ++i) {
			if (tr[i] > 0) { s.push(tr[i]); continue; }

			int x = s.top(); s.pop();
			if (x != 1) {
				int f = s.top();
				sum[f] += min(sum[x], mi[x]);
			}
			else {
				printf("%lld\n", sum[1]);
			}
			sum[x] = 0; book[x] = 0;
		}
	}
	return 0;
}

#include <bits/stdc++.h>
using namespace std;
inline int read() {
	int x = 0; char c = getchar();
	for (; c < '.'; c = getchar()); 
	for (; c > '.'; c = getchar()) x = x * 10 + c - 48;
	return x;
}

const int INF = 0x3f3f3f3f;
const int maxn = 300000 + 7;

struct edge {
	int v, next;
} e[maxn << 1];
int ec, head[maxn];
void add(int u, int v) {
	e[++ec] = (edge) { v, head[u] };
	head[u] = ec;
}
void clear_edge() { 
	memset(head, 0, sizeof(head));
	ec = 0;
}

int n, Q, m, dft;
int dep[maxn], f[maxn][20], dfin[maxn], dfout[maxn], size[maxn];
int a[maxn], b[maxn << 1], c[maxn], sta[maxn], ans[maxn], top, cnt, ncnt, dd;
bool book[maxn];

void dfs1(int x) {
	size[x] = 1;
	dfin[x] = ++dft;
	for (int i = 1; i < 20; ++i) f[x][i] = f[f[x][i - 1]][i - 1];
	for (int i = head[x]; i; i = e[i].next) {
		int v = e[i].v;
		if (v == f[x][0]) continue ;
		dep[v] = dep[x] + 1;
		f[v][0] = x;
		dfs1(v);
		size[x] += size[v];
	}
	dfout[x] = ++dft;
}

int lca(int x, int y) {
	if (dep[x] < dep[y]) swap(x, y);
	int k = dep[x] - dep[y];
	for (int i = 0; k; k >>= 1, ++i) 
		if (k & 1) x = f[x][i];
	if (x == y) return x;
	for (int i = 19; ~i; --i) 
		if (f[x][i] != f[y][i])
			x = f[x][i], y = f[y][i];
	return f[x][0];
}

int dis(int x, int y) {
	return dep[x] + dep[y] - 2 * dep[lca(x, y)];
}

bool cmp1(int x, int y) {
	return dfin[x] < dfin[y];
}

bool cmp2(int x, int y) {
	return (x > 0 ? dfin[x] : dfout[-x]) < (y > 0 ? dfin[y] : dfout[-y]);
}

struct data {
	int w, c;
	bool operator > (const data & i) {
		return w == i.w ? c > i.c : w > i.w;
	}
} g[maxn];

void dfs2(int x) {
	if (book[x]) g[x] = (data) {0, x};
	else g[x] = (data) { INF, 0 };
	for (int i = head[x]; i; i = e[i].next) {
		int v = e[i].v;
		dfs2(v);
		if (g[x] > (data) { g[v].w + dep[v] - dep[x], g[v].c })
			g[x] = (data) { g[v].w + dep[v] - dep[x], g[v].c };
	}
}

void dfs3(int x) {
	for (int i = head[x]; i; i = e[i].next) {
		int v = e[i].v;
		if (g[v] > (data) { g[x].w + dep[v] - dep[x], g[x].c }) 
			g[v] = (data) { g[x].w + dep[v] - dep[x], g[x].c };
		dfs3(v);
	}
}

void dfs4(int x) {
	int c = g[x].c, d;
	for (int i = head[x]; i; i = e[i].next) {
		int v = e[i].v;
		d = g[v].c;
		if (c != d) {
			dd = dep[d] - (dis(c, d) - (c < d)) / 2;
			for (int j = 19; ~j; --j) 
				if (dep[f[v][j]] >= dd) 
					v = f[v][j];
			ans[c] -= size[v];
			ans[d] += size[v];
		}
		dfs4(e[i].v);
	}
}

void dfs5(int x) {
	for (int i = head[x]; i; i = e[i].next) dfs5(e[i].v);
	head[x] = ans[x] = book[x] = 0;
}

int main() {
	n = read();
	for (int i = 1, u, v; i < n; ++i) {
		u = read(); v = read();
		add(u, v); add(v, u);
	}
	dep[1] = 1;
	dfs1(1);
	clear_edge();
	Q = read();
	while (Q--) {
		m = read();
		for (int i = 1; i <= m; ++i) {
			c[i] = read();
			book[c[i]] = true;
			a[i] = c[i];
		}


		sort(a + 1, a + 1 + m, cmp1);
		cnt = 0;
		b[++cnt] = a[1];
		b[++cnt] = -a[1];
		for (int i = 2; i <= m; ++i) {
			int x = lca(a[i - 1], a[i]);
			b[++cnt] = x;
			b[++cnt] = -x;
			b[++cnt] = a[i];
			b[++cnt] = -a[i];
		}
		sort(b + 1, b + 1 + cnt, cmp2);
		ncnt = 0;
		for (int i = 1; i <= cnt; ++i) {
			if (b[i] != b[i - 1]) b[++ncnt] = b[i];
		}
		for (int i = 1; i <= ncnt; ++i) {
			if (b[i] > 0) sta[++top] = b[i];
			else add(sta[top - 1], sta[top]), top--;
		}
		dfs2(b[1]); dfs3(b[1]);
		ans[g[b[1]].c] = size[1];
		dfs4(b[1]);
		for (int i = 1; i <= m; ++i) printf("%d ", ans[c[i]]); putchar('\n');
		dfs5(b[1]);
	}
	return 0;
}
